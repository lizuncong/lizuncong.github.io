index.jsx
```jsx
import React, { memo, useState, useRef, useEffect } from 'react';
import classnames from 'classnames/bind';
import Icon from '@/components/Icon';
import style from './index.scss';

const cs = classnames.bind(style);

// 触发onchange的条件:
// 1.键盘收起
// 2.点击输入框以外的区域
const Index = memo(({ placeholder, onChange, value, maxLength = 100, rows = 2 }) => {
  const [text, setText] = useState('');
  const inputRef = useRef(null);
  const inputMaskRef = useRef(null);

  useEffect(() => {
    inputMaskRef.current.bodyHeight = document.body.getBoundingClientRect().height;
  }, []);
  useEffect(() => {
    const onResize = () => {
      const currentBodyHeight = document.body.getBoundingClientRect().height;
      if (currentBodyHeight < inputMaskRef.current.bodyHeight) {
        console.log('键盘弹出了。。。。');
      } else if (inputMaskRef.current.style.display === 'block') {
        // 键盘收起，并且输入框还是可见的情况下，要隐藏输入框
        console.log('键盘收起了。。。。');
        inputMaskRef.current.style.display = 'none';
        inputRef.current.blur();
        onChange(text);
      }
    };
    window.addEventListener('resize', onResize);
    return () => window.removeEventListener('resize', onResize);
  }, [onChange, text]);

  useEffect(() => {
    if (value !== text) {
      setText(value);
    }
  }, [value]);

  const onInputOk = () => {
    inputMaskRef.current.style.display = 'none';
    inputRef.current.blur();
    onChange(text);
  };

  return (
    <React.Fragment>
      <div
        className={cs('row')}
        onClick={() => {
          inputMaskRef.current.style.display = 'block';
          inputRef.current.focus();
        }}
      >
        { value || placeholder}
      </div>
      <div
        className={cs('mask')}
        ref={inputMaskRef}
        id="input-mask"
        onClick={(e) => {
          console.log('e...', e.target.id);
          if (e.target.id === 'input-mask') {
            onInputOk();
          } else {
            // 输入框一旦失去焦点，就一定会导致键盘收起，
            // 本来打算如果点击的是input-wrap整个区域时，强制让输入框聚焦，就不让键盘收起的
            // inputRef.current.focus();
          }
        }}
      >
        <div
          ref={inputMaskRef}
          className={cs('input-wrap')}
        >
          <textarea
            ref={inputRef}
            value={text}
            maxLength={maxLength}
            rows={rows}
            placeholder={placeholder || '请输入备注'}
            className={cs('input')}
            onChange={e => setText(e.target.value)}
          />
          <div className={cs('bottom')}>
            <Icon type="okr-aite" className={cs('icon')} />
            <Icon
              type="okr-fasong"
              className={cs('icon', 'send')}
              onClick={() => {
                onInputOk();
              }}
            />
          </div>
        </div>
      </div>
    </React.Fragment>
  );
});

export default Index;

```

使用：
```jsx
import React, { useEffect, useState, useCallback } from 'react';
import Input from './input';

const View = React.memo((props) => {
  const [inputText, setInputText] = useState('');
  const onInputChange = useCallback((value) => { setInputText(value); }, []);
  return (
    <div
      className={cs('container')}
    >
        <Input
          placeholder="占位的"
          value={inputText}
          onChange={onInputChange}
        />
    </div>
  );
});
export default View;
```
