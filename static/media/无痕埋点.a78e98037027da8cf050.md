class ReactTrackSdk {
  constructor(config) {
    this.config = {
      reportUrl: config.reportUrl,
      userId: config.userId || 'unknown',
      ignoreSelectors: config.ignoreSelectors || ['[data-track-ignore]'],
      batchSize: 1,
      eventQueue: [],
      // 曝光配置：避免重复曝光（存储已曝光元素ID）
      exposedElements: new Set(),
      // 可见性检测阈值（元素30%可见即触发，可配置）
      exposeThreshold: config.exposeThreshold || 0.3
    };
    this.init();
  }

  init() {
    this.bindGlobalEvents();
    this.initExposeObserver(); // 初始化曝光检测器
    this.scanDomForExpose(); // 扫描已有DOM元素
    this.initBatchReport();
    this.listenDomUpdate(); // 监听DOM动态更新
  }

  // 核心：初始化曝光检测器（IntersectionObserver）
  initExposeObserver() {
    this.exposeObserver = new IntersectionObserver(
      (entries) => {
        entries.forEach(entry => {
          // 1. 元素可见且未被曝光过
          if (entry.isIntersecting && !this.config.exposedElements.has(entry.target.dataset.trackExposeId)) {
            const exposeConfig = JSON.parse(entry.target.dataset.trackExpose);
            // 2. 上报曝光事件
            this.reportEvent({
              eventType: 'expose',
              eventName: exposeConfig.eventName,
              pageUrl: window.location.href,
              userId: this.config.userId,
              elementInfo: {
                id: entry.target.dataset.trackExposeId,
                tagName: entry.target.tagName.toLowerCase(),
                text: entry.target.innerText.trim() || '无文本'
              },
              // 扩展：曝光时元素可见比例
              visibleRatio: entry.intersectionRatio.toFixed(2)
            });
            // 3. 标记为已曝光，避免重复上报
            this.config.exposedElements.add(entry.target.dataset.trackExposeId);
          }
        });
      },
      {
        root: null, // 视口作为根元素
        rootMargin: '0px', // 无额外边距
        threshold: this.config.exposeThreshold // 可见阈值
      }
    );
  }

  // 扫描页面中带曝光标识的DOM元素，加入检测
  scanDomForExpose() {
    // 匹配所有带data-track-expose的元素
    const exposeElements = document.querySelectorAll('[data-track-expose]');
    exposeElements.forEach(el => {
      // 给元素生成唯一ID（避免重复）
      if (!el.dataset.trackExposeId) {
        el.dataset.trackExposeId = `expose_${Date.now()}_${Math.random().toString(36).slice(2)}`;
      }
      // 加入曝光检测
      this.exposeObserver.observe(el);
    });
  }

  // 监听DOM动态更新（如列表渲染、弹窗出现），重新扫描曝光元素
  listenDomUpdate() {
    // 使用MutationObserver监听body内DOM变化
    const observer = new MutationObserver((mutations) => {
      mutations.forEach(mutation => {
        // 新增DOM节点时，重新扫描曝光元素
        if (mutation.addedNodes.length) {
          this.scanDomForExpose();
        }
      });
    });
    console.log('document.body...', document.body)
    observer.observe(document.body, {
      childList: true, // 监听子节点变化
      subtree: true // 深度监听（子树内变化也触发）
    });
  }

  // 点击事件处理（复用原有逻辑）
  bindGlobalEvents() {
    document.addEventListener('click', (e) => {
      const target = this.getTrackTarget(e.target);
      if (target?.dataset.trackClick) {
        this.reportEvent({
          eventType: 'click',
          eventName: JSON.parse(target.dataset.trackClick).eventName,
          pageUrl: window.location.href,
          userId: this.config.userId,
          elementInfo: {
            tagName: target.tagName.toLowerCase(),
            text: target.innerText.trim()
          }
        });
      }
    }, true);
  }

  // 事件上报与批量处理（复用原有逻辑）
  reportEvent(event) {
    const reportEvent = {
      ...event,
      timestamp: Date.now(),
      userAgent: navigator.userAgent
    };
    this.config.eventQueue.push(reportEvent);
    if (this.config.eventQueue.length >= this.config.batchSize) {
      this.sendBatchEvents();
    }
  }

  sendBatchEvents() {
    if (!this.config.eventQueue.length) return;
    fetch(this.config.reportUrl, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(this.config.eventQueue)
    })
      .then(res => res.json())
      .then(() => this.config.eventQueue = [])
      .catch(err => console.error('Track report failed:', err));
  }

  initBatchReport() {
    window.addEventListener('beforeunload', () => this.config.eventQueue.length && this.sendBatchEvents());
    setInterval(() => this.config.eventQueue.length && this.sendBatchEvents(), 30000);
  }

  getTrackTarget(element) {
    let current = element;
    while (current && current !== document) {
      if (current.dataset.trackClick || current.dataset.trackExpose) return current;
      current = current.parentElement;
    }
    return null;
  }
}

window.ReactTrackSdk = ReactTrackSdk;
